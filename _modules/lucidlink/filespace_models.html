class SyncMode(Enum):
    """Controls whether changes are automatically synchronized to the hub on close.

    SYNC_ALL: (default) Automatically call sync_all() before unlinking filespaces.
    SYNC_NONE: Do not sync automatically — caller must call sync_all() explicitly.
    """

    SYNC_NONE = "none"
    SYNC_ALL = "all"

    Attributes:

        name: Human-readable filespace name
        created: Creation timestamp (ISO 8601)

class DaemonStatus:
    """
    Daemon status information.

    Returned by Daemon.get_status().

    Attributes:
        is_running: Whether the daemon is currently running
        is_authenticated: Whether the daemon has authenticated to a workspace

    is_running: bool

    is_linked: bool
    root_path: str
